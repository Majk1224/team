### 深浅拷贝

==数组拷贝：==

    // [...arr], concat(), slice()
    var arr1 = [1, 2, 3, {
        list: [4,5,6,7]
    }]
    //...只能拷贝一层
    var arr2 = [...arr1];
    arr1[3].list.push(8);
    console.log('arr1 === arr2', arr1 === arr2, arr1.list === arr2.list, arr2[3].list);
